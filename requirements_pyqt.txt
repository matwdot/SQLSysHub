# Dependências opcionais para drivers de banco de dados

# Firebird (alternativo)

# pyodbc

# PostgreSQL
# psycopg2-binary

# MySQL
# mysql-connector-python

# Para instalar PyQt5 e Firebird (recomendado):
# pip install PyQt5 firebirdsql passlib

# pip install PyQt5 firebirdsql passlib fdb pyodbc psycopg2-binary mysql-connector-python
